58 template <typename Scalar, int nq>
59 Eigen::Matrix<Scalar, nq + nq, 1> euler_step(Model<Scalar, nq>& model,
60 const Eigen::Matrix<Scalar, nq, 1>& qkm1,
61 const Eigen::Matrix<Scalar, nq, 1>& pkm1,
62 const Eigen::Matrix<Scalar, nq, 1>& ukm1,
63 const float& dt,
64 const std::vector<std::string> active_constraints = {}) {
65 // Compute the dynamics of the system
66 forward_dynamics(model, qkm1, pkm1, ukm1, active_constraints);
67
68 // Perform the q update: qk = qkm1 + dt*dqm1dt
69 Eigen::Matrix<Scalar, nq, 1> qk = qkm1 + dt * model.data.dx_dt.head(nq);
70
71 // Perform the p update: pk = pkm1 + dt*dpm1dt
72 Eigen::Matrix<Scalar, nq, 1> pk = pkm1 + dt * model.data.dx_dt.tail(nq);
73
74 // Store the new state
75 Eigen::Matrix<Scalar, nq + nq, 1> xk;
76 xk << qk, pk;
77 return xk;
78 }

90 const Eigen::Matrix<Scalar, nq, 1>& qkm1,

93 const float& dt,

95 // Compute the mass matrix, which can be used for q update
96 Eigen::Matrix<Scalar, nq, nq> Mkm1 = mass_matrix(model, qkm1);
97
98 // Perform the q update: qk = qkm1 + dt*dqm1dt
99 Eigen::Matrix<Scalar, nq, 1> qk = qkm1 + dt * (Mkm1.inverse() * pkm1);

102 forward_dynamics(model, qk, pkm1, ukm1, active_constraints);
103
104 // Perform the p update: pk = pkm1 + dt*dpm1dt

106

121 template <typename Scalar, int nq>
122 Eigen::Matrix<Scalar, nq + nq, 1> RK4(Model<Scalar, nq>& model,
123 const Eigen::Matrix<Scalar, nq, 1>& qkm1,
124 const Eigen::Matrix<Scalar, nq, 1>& pkm1,
125 const Eigen::Matrix<Scalar, nq, 1>& ukm1,
126 const float& dt,
127 const std::vector<std::string> active_constraints = {}) {
128 Eigen::Matrix<Scalar, nq + nq, 1> k1, k2, k3, k4, xk;
129 // Compute the mass matrix, which can be used for q update
130 k1 = forward_dynamics(model, qkm1, pkm1, ukm1, active_constraints);
131 k2 = forward_dynamics(model,
132 Eigen::Matrix<Scalar, nq, 1>(qkm1 + 0.5 * dt * k1.head(nq)),
133 Eigen::Matrix<Scalar, nq, 1>(pkm1 + 0.5 * dt * k1.tail(nq)),
134 ukm1,
135 active_constraints);
136 k3 = forward_dynamics(model,
137 Eigen::Matrix<Scalar, nq, 1>(qkm1 + 0.5 * dt * k2.head(nq)),
138 Eigen::Matrix<Scalar, nq, 1>(pkm1 + 0.5 * dt * k2.tail(nq)),
139 ukm1,
140 active_constraints);
141 k4 = forward_dynamics(model,
142 Eigen::Matrix<Scalar, nq, 1>(qkm1 + dt * k3.head(nq)),
143 Eigen::Matrix<Scalar, nq, 1>(pkm1 + dt * k3.tail(nq)),
144 ukm1,
145 active_constraints);
146 // Store the new state
147 xk << qkm1 + dt / 6 * (k1.head(nq) + 2 * k2.head(nq) + 2 * k3.head(nq) + k4.head(nq)),
148 pkm1 + dt / 6 * (k1.tail(nq) + 2 * k2.tail(nq) + 2 * k3.tail(nq) + k4.tail(nq));
149 return xk;
150 }

160 template <typename Scalar, int nq>
161 Eigen::Matrix<Scalar, nq + nq, 1> integrate(Model<Scalar, nq>& model,
162 const Eigen::Matrix<Scalar, nq, 1>& qkm1,
163 const Eigen::Matrix<Scalar, nq, 1>& pkm1,
164 const Eigen::Matrix<Scalar, nq, 1>& ukm1,
165 const float& dt,
166 const IntegrationMethod& integration_type,
167 const std::vector<std::string> active_constraints) {
168 // Perform the integration step based on the integration method
169 switch (integration_type) {
170 case IntegrationMethod::EULER: return euler_step(model, qkm1, pkm1, ukm1, dt, active_constraints);
171 case IntegrationMethod::SYMPLECTIC_EULER:
172 return symplectic_euler_step(model, qkm1, pkm1, ukm1, dt, active_constraints);
173 case IntegrationMethod::RK4: return RK4(model, qkm1, pkm1, ukm1, dt, active_constraints);
174 default: throw std::runtime_error("Unknown integration method.");
175 }
176 }

211 template <typename Scalar, int nq, typename... IntMethod>
212 SolverResults<Scalar, nq> solver(Model<Scalar, nq>& model,
213 Eigen::Matrix<Scalar, nq, 1> qk,
214 Eigen::Matrix<Scalar, nq, 1> pk,
215 Eigen::Matrix<Scalar, nq, 1> u,
216 SolverParams<Scalar, nq>& params) {
217 // Create a struct for solver results
218 SolverResults<Scalar, nq> results;
219
220 // Ensure that the time span is valid
221 if (params.tspan.first >= params.tspan.second) {
222 throw std::invalid_argument("The time span is invalid.");
223 }
224 // Ensure that the time step is valid
225 if (params.dt <= 0 || params.dt >= params.tspan.second - params.tspan.first) {
226 throw std::invalid_argument("The time step is invalid.");
227 }
228
229 // For each time step in the time span, integrate the robot dynamics.
230 const int n_steps = (params.tspan.second - params.tspan.first) / params.dt + 2;
231 for (int k = 0; k < n_steps; k++) {
232 // Compute the current time
233 const Scalar t = params.tspan.first + k * params.dt;
234 results.time.push_back(t);
235 // TODO: Compute the control input
236 // u = controller(model,qk,pk,t);
237 results.u_history.push_back(u);
238 // Perform the integration step
239 results.x_history.push_back(
240 integrate(model, qk, pk, u, params.dt, params.integration_method, params.active_constraints));
241 // Update the initial state for the next integration step
242 qk = results.x_history.back().head(nq);
243 pk = results.x_history.back().tail(nq);
244 // Check if event has occurred
245 if (params.event_detection != nullptr) {
246 if (params.event_detection(model, qk, pk)) {
247 std::cout << "Event occurred at time: " << t << std::endl;
248 if (params.event_is_terminal) {
249 break;
250 }
251 }
252 }
253 }
254 return results;
255 }
