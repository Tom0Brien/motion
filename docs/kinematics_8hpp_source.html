46 template <typename Scalar, int nq>
47 std::vector<Eigen::Transform<Scalar, 3, Eigen::Isometry>> forward_kinematics(
48 Model<Scalar, nq>& model,
49 const Eigen::Matrix<Scalar, nq, 1>& q) {
50 model.data.forward_kinematics.resize(model.n_links, Eigen::Transform<Scalar, 3, Eigen::Isometry>::Identity());
51 for (auto const link : model.links) {
52 model.data.forward_kinematics[link.idx] = link.joint.parent_transform;
53 if (link.joint.idx != -1) {
54 model.data.forward_kinematics[link.idx] =
55 model.data.forward_kinematics[link.idx] * link.joint.get_joint_transform(q[link.joint.idx]);
56 }
57 if (link.parent != -1) {
58 model.data.forward_kinematics[link.idx] =
59 model.data.forward_kinematics[link.parent] * model.data.forward_kinematics[link.idx];
60 }
61 }
62 return model.data.forward_kinematics;
63 }

75 template <typename Scalar, int nq, typename TargetLink>
76 Eigen::Transform<Scalar, 3, Eigen::Isometry> forward_kinematics(const Model<Scalar, nq>& model,
77 const Eigen::Matrix<Scalar, nq, 1>& q,
78 const TargetLink& target_link) {
79 // Get the target link
80 const int target_link_idx = get_link_idx(model, target_link);
81 Link<Scalar> current_link = model.links[target_link_idx];
82
83 // Build kinematic tree from target_link {t} to base {b}
84 Eigen::Transform<Scalar, 3, Eigen::Isometry> Htb = Eigen::Transform<Scalar, 3, Eigen::Isometry>::Identity();
85 while (current_link.idx != model.base_link_idx) {
86 Eigen::Transform<Scalar, 3, Eigen::Isometry> H = Eigen::Transform<Scalar, 3, Eigen::Isometry>::Identity();
87 if (current_link.joint.idx != -1) {
88 // Transform the joint frame by joint value (Hpt)
89 H = current_link.joint.get_joint_transform(q[current_link.joint.idx]);
90 }
91 Htb = Htb * H.inverse();
92 // Apply inverse joint transform as we are going back up tree
93 Htb = Htb * current_link.joint.parent_transform.inverse();
94 // Move up the tree to parent link
95 current_link = model.links[current_link.parent];
96 }
97 // Return transform Hbt, which is transform from base {b} to target {t}
98 return Htb.inverse();
99 }

113 template <typename Scalar, int nq, typename TargetLink, typename SourceLink>
114 Eigen::Transform<Scalar, 3, Eigen::Isometry> forward_kinematics(const Model<Scalar, nq>& model,
115 const Eigen::Matrix<Scalar, nq, 1>& q,
116 const TargetLink& target_link,
117 const SourceLink& source_link) {
118 const int target_link_idx = get_link_idx(model, target_link);
119 const int source_link_idx = get_link_idx(model, source_link);
120
121 // Build kinematic tree from source {s} to base {b} frame
122 Eigen::Transform<Scalar, 3, Eigen::Isometry> Hbs = forward_kinematics(model, q, source_link_idx);
123 Eigen::Transform<Scalar, 3, Eigen::Isometry> Hsb = Hbs.inverse();
124 // Build kinematic tree from base {b} to target {t} frame
125 Eigen::Transform<Scalar, 3, Eigen::Isometry> Hbt = forward_kinematics(model, q, target_link_idx);
126 // Compute transform between source {s} and target {t} frames
127 Eigen::Transform<Scalar, 3, Eigen::Isometry> Hst = Hsb * Hbt;
128 return Hst;
129 }

140 std::vector<Eigen::Transform<Scalar, 3, Eigen::Isometry>> forward_kinematics_com(
141 Model<Scalar, nq>& model,
142 const Eigen::Matrix<Scalar, nq, 1>& q) {
143 // Compute forward kinematics for all the links
144 forward_kinematics(model, q);
145 // Compute transform to centre of mass of all the forward kinematics results

165 template <typename Scalar, int nq, typename TargetLink, typename SourceLink = int>
166 Eigen::Transform<Scalar, 3, Eigen::Isometry> forward_kinematics_com(const Model<Scalar, nq>& model,
167 const Eigen::Matrix<Scalar, nq, 1>& q,
168 const TargetLink& target_link,
169 const SourceLink& source_link = 0) {
170 // Compute forward kinematics from source {b} to target {t}
171 Eigen::Transform<Scalar, 3, Eigen::Isometry> Hst = forward_kinematics(model, q, target_link, source_link);
172 // Compute forward kinematics from source {s} to CoM {c}
173 const int target_link_idx = get_link_idx(model, target_link);
174 Eigen::Transform<Scalar, 3, Eigen::Isometry> Hsc = Hst * model.links[target_link_idx].centre_of_mass;
175 // Return transform from source {s} to CoM {c}
176 return Hsc;
177 }

229 template <typename Scalar, int nq, typename TargetLink>
230 Eigen::Matrix<Scalar, 6, nq> geometric_jacobian(Model<Scalar, nq>& model,
231 const Eigen::Matrix<Scalar, nq, 1>& q,
232 const TargetLink& target_link) {
233 // Compute the transform between base {b} and all the links {i} in the kinematic chain
234 forward_kinematics(model, q);
235
236 // Get the target link from the model
237 const int target_link_idx = get_link_idx(model, target_link);
238
239 // Get the displacement of the target link {t} in the base link frame {b}
240 Eigen::Matrix<Scalar, 3, 1> rTBb = model.data.forward_kinematics[target_link_idx].translation();
241
242 // Initialize variables
243 model.data.J.setZero();
244 Eigen::Matrix<Scalar, 3, 1> zIBb;
245 Eigen::Matrix<Scalar, 3, 1> rIBb;
246 Link<Scalar> current_link = model.links[target_link_idx];
247 while (current_link.idx != model.base_link_idx) {
248 if (current_link.joint.idx != -1) {
249 // Axis of the current joint rotated into the base frame {b}
250 zIBb = model.data.forward_kinematics[current_link.idx].linear() * current_link.joint.axis;
251 // Compute the displacement of the current link {i} from the base link frame {b}
252 rIBb = model.data.forward_kinematics[current_link.idx].translation();
253 if (current_link.joint.type == JointType::PRISMATIC) {
254 model.data.J.block(0, current_link.joint.idx, 3, 1) = zIBb;
255 model.data.J.block(3, current_link.joint.idx, 3, 1) = Eigen::Matrix<Scalar, 3, 1>::Zero();
256 }
257 else if (current_link.joint.type == JointType::REVOLUTE) {
258 model.data.J.block(0, current_link.joint.idx, 3, 1) = (zIBb).cross(rTBb - rIBb);
259 model.data.J.block(3, current_link.joint.idx, 3, 1) = zIBb;
260 }
261 }
262 // Move up the tree to parent towards the base
263 current_link = model.links[current_link.parent];
264 }
265 return model.data.J;
266 }

278 template <typename Scalar, int nq, typename TargetLink>
279 Eigen::Matrix<Scalar, 6, nq> geometric_jacobian_com(const Model<Scalar, nq>& model,
280 const Eigen::Matrix<Scalar, nq, 1>& q,
281 const TargetLink& target_link) {
282 // Get the target link from the model
283 const int target_link_idx = get_link_idx(model, target_link);
284 Link<Scalar> current_link = model.links[target_link_idx];
285
286 // Get the base link from the model
287 auto base_link = model.links[model.base_link_idx];
288
289 // Compute the displacement of the target link {t} in the base link frame {b}
290 Eigen::Matrix<Scalar, 3, 1> rTcBb = forward_kinematics_com(model, q, target_link, base_link.name).translation();
291
292 // Initialize the geometric jabobian matrix with zeros
293 Eigen::Matrix<Scalar, 6, nq> J = Eigen::Matrix<Scalar, 6, nq>::Zero();
294 while (current_link.idx != model.base_link_idx) {
295 auto current_joint = current_link.joint;
296 if (current_joint.idx != -1) {
297 // Compute the transform between base {b} and the current link {i}
298 auto Hbi = forward_kinematics(model, q, current_link.name, base_link.name);
299 // Axis of the current joint rotated into the base frame {b}
300 auto zIBb = Hbi.linear() * current_joint.axis;
301 // Compute the displacement of the current link {i} from the base link frame {b}
302 auto rIBb = Hbi.translation();
303 if (current_joint.type == JointType::PRISMATIC) {
304 J.block(0, current_joint.idx, 3, 1) = zIBb;
305 J.block(3, current_joint.idx, 3, 1) = Eigen::Matrix<Scalar, 3, 1>::Zero();
306 }
307 else if (current_joint.type == JointType::REVOLUTE) {
308 J.block(0, current_joint.idx, 3, 1) = (zIBb).cross(rTcBb - rIBb);
309 J.block(3, current_joint.idx, 3, 1) = zIBb;
310 }
311 }
312 // Move up the tree to parent towards the base
313 current_link = model.links[current_link.parent];
314 }
315 return J;
316 }

328 template <typename Scalar, int nq, typename TargetLink>
329 Eigen::Matrix<Scalar, 6, nq> analytical_jacobian(Model<Scalar, nq>& model,
330 const Eigen::Matrix<Scalar, nq, 1>& q,
331 const TargetLink& target_link) {
332 // Compute the geometric jacobian
333 model.data.J = geometric_jacobian(model, q, target_link);
334
335 auto Rbt = rotation(model, q, target_link);
336 auto rpy = Rbt.eulerAngles(2, 1, 0);
337 Scalar phi = rpy(0);
338 Scalar theta = rpy(1);
339 Scalar psi = rpy(2);
340
341 Eigen::Matrix<Scalar, 6, 6> E = Eigen::Matrix<Scalar, 6, 6>::Identity();
342 E.block(3, 3, 3, 3) << 1, 0, -sin(theta), 0, cos(phi), cos(theta) * sin(phi), 0, -sin(phi),
343 cos(theta) * cos(phi);
344
345 model.data.Ja = E * model.data.J;
346 return model.data.Ja;
347 }

360 template <typename Scalar, int nq, typename SourceLink = int>
361 Eigen::Matrix<Scalar, 3, 1> centre_of_mass(const Model<Scalar, nq>& model,
362 const Eigen::Matrix<Scalar, nq, 1>& q,
363 const SourceLink& source_link = 0) {
364 // Initialize the centre of mass as zero
365 Eigen::Matrix<Scalar, 3, 1> rCSs = Eigen::Matrix<Scalar, 3, 1>::Zero();
366 for (auto link : model.links) {
367 // Compute the transform from the source link to the CoM of the link
368 Eigen::Transform<Scalar, 3, Eigen::Isometry> Hsc = forward_kinematics_com(model, q, link.name, source_link);
369 // Compute the centre of mass of the link
370 rCSs = rCSs + Hsc.translation() * link.mass;
371 }
372 // Compute the centre of mass from the source link
373 return rCSs / model.mass;
374 }
