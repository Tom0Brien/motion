21 inline Eigen::Matrix<Scalar, 3, 1> rotation_to_rpy(const T& mat) {
22 using std::atan2;
23 using std::sqrt;
24 // Eigen euler angles and with better range
25 return Eigen::Matrix<Scalar, 3, 1>(
26 // Roll
27 atan2(mat(2, 1), mat(2, 2)),
28 // Pitch
29 atan2(-mat(2, 0), sqrt(mat(2, 1) * mat(2, 1) + mat(2, 2) * mat(2, 2))),

187 template <typename Scalar>
188 Eigen::Matrix<Scalar, 6, 1> homogeneous_error(const Eigen::Transform<Scalar, 3, Eigen::Isometry>& H1,
189 const Eigen::Transform<Scalar, 3, Eigen::Isometry>& H2) {
190
191 Eigen::Matrix<Scalar, 3, 3> R1, R2, Re;
192 Eigen::Matrix<Scalar, 3, 1> d1, d2, eps;
193 Eigen::Matrix<Scalar, 6, 1> e;
194 Scalar eps_norm, t;
195
196 // Break out values
197 R1 = H1.rotation();
198 R2 = H2.rotation();
199 d1 = H1.translation();
200 d2 = H2.translation();
201
202 // Orientation error
203 Re = R1 * R2.transpose();
204
205 // Assign linear error
206 e.head(3) = d1 - d2;
207
208 // Extract diagonal and trace
209 t = Re.trace();
210
211 // Build l variable, and calculate norm
212 eps << Re(2, 1) - Re(1, 2), Re(0, 2) - Re(2, 0), Re(1, 0) - Re(0, 1);
213 eps_norm = eps.norm();
214
215 if (t > -.99 || eps_norm > 1e-10) {
216 if (eps_norm < 1e-3) {
217 e.tail(3) = (0.75 - t / 12) * eps;
218 }
219 else {
220 e.tail(3) = (atan2(eps_norm, t - 1) / eps_norm) * eps;
221 }
222 }
223 else {
224 e.tail(3) = 1.570796326794897 * (Re.diagonal().array() + 1);
225 }
226 return e;
227 }
