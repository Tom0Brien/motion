178 template <typename Scalar, int nq>
179 Scalar cost(const Eigen::Matrix<Scalar, nq, 1>& q,
180 Model<Scalar, nq>& model,
181 const std::string& target_link_name,
182 const std::string& source_link_name,
183 const Eigen::Transform<Scalar, 3, Eigen::Isometry>& desired_pose,
184 const Eigen::Matrix<Scalar, nq, 1>& q0,
185 Eigen::Matrix<Scalar, nq, 1>& gradient,
186 const InverseKinematicsOptions<Scalar, nq>& options) {
187
188 // Compute the current pose
189 Eigen::Transform<Scalar, 3, Eigen::Isometry> current_pose =
190 forward_kinematics(model, q, target_link_name, source_link_name);
191
192 // Compute the pose error
193 Eigen::Matrix<Scalar, 6, 1> pose_error = homogeneous_error(current_pose, desired_pose);
194
195 // Compute the cost: q^T*W*q + (k(q) - x*)^TK*(k(q) - x*))), where k(q) is the current pose, x* is the desired
196 // pose, and W and K are the weighting matrices
197 Eigen::Matrix<Scalar, 1, 1> cost =
198 0.5 * pose_error.transpose() * options.K * pose_error + 0.5 * (q - q0).transpose() * options.W * (q - q0);
199
200 // Compute the gradient of the cost function
201 Eigen::Matrix<Scalar, 6, nq> J = geometric_jacobian(model, q, target_link_name);
202 gradient = J.transpose() * options.K * pose_error + options.W * (q - q0);
203
204 return cost(0);
205 }

219 template <typename Scalar, int nq>
220 Scalar cost_autodiff(const Eigen::Matrix<Scalar, nq, 1>& q,
221 Model<Scalar, nq>& model,
222 const std::string& target_link_name,
223 const std::string& source_link_name,
224 const Eigen::Transform<Scalar, 3, Eigen::Isometry>& desired_pose,
225 const Eigen::Matrix<Scalar, nq, 1>& q0,
226 Eigen::Matrix<Scalar, nq, 1>& gradient,
227 const InverseKinematicsOptions<Scalar, nq>& options) {
228
229 using ADScalar = Eigen::AutoDiffScalar<Eigen::Matrix<Scalar, nq, 1>>;
230
231 // Cast model to Eigen::AutoDiffScalar
232 Model<ADScalar, nq> model_ad = model.template cast<ADScalar>();
233
234 // Initialize q_ad with derivatives
235 Eigen::Matrix<ADScalar, nq, 1> q_ad;
236 Eigen::Matrix<ADScalar, nq, 1> q0_ad;
237 for (int i = 0; i < nq; ++i) {
238 q_ad[i] = ADScalar(q[i]);
239 q_ad[i].derivatives()[i] = 1.0;
240 q0_ad[i] = ADScalar(q0[i]);
241 }
242
243 // Compute the current pose
244 Eigen::Transform<ADScalar, 3, Eigen::Isometry> current_pose_ad =
245 forward_kinematics(model_ad, q_ad, target_link_name, source_link_name);
246
247 // Cast the desired pose to ADScalar
248 Eigen::Transform<ADScalar, 3, Eigen::Isometry> desired_pose_ad = desired_pose.template cast<ADScalar>();
249
250 // Stack the position and orientation errors
251 Eigen::Matrix<ADScalar, 6, 1> pose_error = homogeneous_error(current_pose_ad, desired_pose_ad);
252
253 Eigen::Matrix<ADScalar, 6, 6> K_ad = options.K.template cast<ADScalar>();
254 Eigen::Matrix<ADScalar, nq, nq> W_ad = options.W.template cast<ADScalar>();
255
256 // Compute the cost: q^T*W*q + (k(q) - x*)^TK*(k(q) - x*))), where k(q) is the current pose, x* is the desired
257 // pose, and W and K are the weighting matrices
258 Eigen::Matrix<ADScalar, 1, 1> cost =
259 0.5 * pose_error.transpose() * K_ad * pose_error + 0.5 * (q_ad - q0_ad).transpose() * W_ad * (q_ad - q0_ad);
260
261 // Extract the gradient from the gradient_ad
262 for (int i = 0; i < nq; ++i) {
263 gradient[i] = cost[0].derivatives()[i];
264 }
265
266 // Return the value of the cost function
267 return Scalar(cost(0).value());
268 }

281 template <typename Scalar, int nq>
282 Eigen::Matrix<Scalar, nq, 1> inverse_kinematics_nlopt(
283 Model<Scalar, nq>& model,
284 const std::string& target_link_name,
285 const std::string& source_link_name,
286 const Eigen::Transform<Scalar, 3, Eigen::Isometry>& desired_pose,
287 const Eigen::Matrix<Scalar, nq, 1> q0,
288 const InverseKinematicsOptions<Scalar, nq>& options) {
289 // Create the NLopt optimizer and set the algorithm
290 nlopt::opt opt(options.algorithm, nq);
291
292 // Wrap the objective function in a lambda function
293 ObjectiveFunction<Scalar, nq> obj_fun =
294 [&](const Eigen::Matrix<Scalar, nq, 1>& q, Eigen::Matrix<Scalar, nq, 1>& grad, void* data) -> Scalar {
295 (void) data; // Unused in this case
296
297 // Compute the cost and gradient
298 Eigen::Matrix<Scalar, nq, 1> gradient;
299 Scalar cost_value = 0.0;
300 if (options.method == InverseKinematicsMethod::NLOPT_AUTODIFF) {
301 cost_value =
302 cost_autodiff(q, model, target_link_name, source_link_name, desired_pose, q0, gradient, options);
303 }
304 else {
305 cost_value = cost(q, model, target_link_name, source_link_name, desired_pose, q0, gradient, options);
306 }
307
308 // Pass the gradient to NLopt
309 for (int i = 0; i < nq; ++i) {
310 grad[i] = gradient(i);
311 }
312
313 return cost_value;
314 };
315 // Set the objective function
316 opt.set_min_objective(eigen_objective_wrapper<Scalar, nq>, &obj_fun);
317
318 // Set the optimization tolerances
319 opt.set_xtol_rel(options.xtol_rel);
320 opt.set_ftol_rel(options.tolerance);
321
322 // Set the maximum number of iterations
323 opt.set_maxeval(options.max_iterations);
324
325 // Convert the initial guess to NLopt format
326 std::vector<Scalar> x_initial(nq);
327 eigen_to_nlopt<Scalar, nq>(q0, x_initial);
328
329 // Find the optimal solution
330 Scalar minf;
331 opt.optimize(x_initial, minf);
332
333 // Convert the optimized solution back to an Eigen
334 Eigen::Matrix<Scalar, nq, 1> optimized_solution(nq);
335 nlopt_to_eigen<Scalar, nq>(x_initial, optimized_solution);
336 return optimized_solution;
337 }

350 template <typename Scalar, int nq>
351 Eigen::Matrix<Scalar, nq, 1> inverse_kinematics_jacobian(
352 Model<Scalar, nq>& model,
353 const std::string& target_link_name,
354 const std::string& source_link_name,
355 const Eigen::Transform<Scalar, 3, Eigen::Isometry>& desired_pose,
356 const Eigen::Matrix<Scalar, nq, 1> q0,
357 const InverseKinematicsOptions<Scalar, nq>& options) {
358
359 // Set the initial guess
360 Eigen::Matrix<Scalar, nq, 1> q_current = q0;
361
362 // Iterate until the maximum number of iterations is reached
363 for (int iteration = 0; iteration < options.max_iterations; ++iteration) {
364
365 // Compute the current pose
366 Eigen::Transform<Scalar, 3, Eigen::Isometry> current_pose =
367 forward_kinematics(model, q_current, target_link_name, source_link_name);
368
369 // Compute the pose error vector
370 Eigen::Matrix<Scalar, 6, 1> pose_error = homogeneous_error(current_pose, desired_pose);
371
372 // Check if the error is within tolerance
373 if (pose_error.norm() < options.tolerance) {
374 break;
375 }
376
377 // Compute the Jacobian matrix
378 Eigen::Matrix<Scalar, 6, nq> J = geometric_jacobian(model, q_current, target_link_name);
379
380 // Compute the change in configuration
381 Eigen::Matrix<Scalar, nq, 1> delta_q =
382 J.completeOrthogonalDecomposition().pseudoInverse() * (-options.step_size * pose_error);
383
384 // Update the current configuration
385 q_current += delta_q;
386 }
387
388 // Return the final configuration
389 return q_current;
390 }

403 template <typename Scalar, int nq>
404 Eigen::Matrix<Scalar, nq, 1> inverse_kinematics_levenberg_marquardt(
405 Model<Scalar, nq>& model,
406 const std::string& target_link_name,
407 const std::string& source_link_name,
408 const Eigen::Transform<Scalar, 3, Eigen::Isometry>& desired_pose,
409 const Eigen::Matrix<Scalar, nq, 1> q0,
410 const InverseKinematicsOptions<Scalar, nq>& options) {
411
412 // Set the initial guess
413 Eigen::Matrix<Scalar, nq, 1> q_current = q0;
414
415 // Initialize the damping factor
416 Scalar lambda = options.initial_damping;
417
418 // Iterate until the maximum number of iterations is reached
419 for (int iteration = 0; iteration < options.max_iterations; ++iteration) {
420
421 // Compute the current pose
422 Eigen::Transform<Scalar, 3, Eigen::Isometry> current_pose =
423 forward_kinematics(model, q_current, target_link_name, source_link_name);
424
425 // Compute the pose error vector
426 Eigen::Matrix<Scalar, 6, 1> pose_error = homogeneous_error(current_pose, desired_pose);
427
428 // Check if the error is within tolerance
429 if (pose_error.norm() < options.tolerance) {
430 break;
431 }
432
433 // Compute the Jacobian matrix
434 Eigen::Matrix<Scalar, 6, nq> J = geometric_jacobian(model, q_current, target_link_name);
435
436 // Compute the Hessian approximation and the gradient
437 Eigen::Matrix<Scalar, nq, nq> H = J.transpose() * J;
438 Eigen::Matrix<Scalar, nq, 1> g = J.transpose() * pose_error;
439
440 // Levenberg-Marquardt update
441 Eigen::Matrix<Scalar, nq, 1> delta_q =
442 (H + lambda * Eigen::Matrix<Scalar, nq, nq>(H.diagonal().asDiagonal())).ldlt().solve(-g);
443
444 // Test the new configuration
445 Eigen::Matrix<Scalar, nq, 1> q_new = q_current + delta_q;
446 Eigen::Transform<Scalar, 3, Eigen::Isometry> new_pose =
447 forward_kinematics(model, q_new, target_link_name, source_link_name);
448
449 // Compute the new error vector
450 Eigen::Matrix<Scalar, 6, 1> new_pose_error;
451 new_pose_error.head(3) = new_pose.translation() - desired_pose.translation();
452 Eigen::AngleAxis<Scalar> new_rot_error(new_pose.rotation().transpose() * desired_pose.rotation());
453 new_pose_error.tail(3) = new_rot_error.angle() * new_rot_error.axis();
454
455 // Check if the new error is smaller than the old error
456 if (new_pose_error.norm() < pose_error.norm()) {
457 // Accept the new configuration
458 q_current = q_new;
459 lambda *= options.damping_decrease_factor;
460 }
461 else {
462 // Reject the new configuration and increase the damping factor
463 lambda *= options.damping_increase_factor;
464 }
465 }
466
467 // Return the final configuration
468 return q_current;
469 }

482 template <typename Scalar, int nq>
483 Eigen::Matrix<Scalar, nq, 1> inverse_kinematics_pso(
484 Model<Scalar, nq>& model,
485 const std::string& target_link_name,
486 const std::string& source_link_name,
487 const Eigen::Transform<Scalar, 3, Eigen::Isometry>& desired_pose,
488 const Eigen::Matrix<Scalar, nq, 1> q0,
489 const InverseKinematicsOptions<Scalar, nq>& options) {
490 // Particle Swarm Optimization parameters
491 int num_particles = options.num_particles;
492 Scalar omega = options.omega;
493 Scalar c1 = options.c1;
494 Scalar c2 = options.c2;
495
496 // Initialize particles and velocities
497 std::vector<Eigen::Matrix<Scalar, nq, 1>> particles(num_particles, q0);
498 std::vector<Eigen::Matrix<Scalar, nq, 1>> velocities(num_particles, Eigen::Matrix<Scalar, nq, 1>::Zero());
499 std::vector<Scalar> fitness_values(num_particles, std::numeric_limits<Scalar>::max());
500 Eigen::Matrix<Scalar, nq, 1> best_global_position = q0;
501 Scalar best_global_fitness = std::numeric_limits<Scalar>::max();
502
503 // Initialize particles
504 for (int i = 0; i < num_particles; ++i) {
505 particles[i] = q0 + options.init_position_scale * Eigen::Matrix<Scalar, nq, 1>::Random();
506 }
507
508 // Main optimization loop
509 for (int iteration = 0; iteration < options.max_iterations; ++iteration) {
510 for (int i = 0; i < num_particles; ++i) {
511 // Evaluate the fitness of the particle
512 Eigen::Transform<Scalar, 3, Eigen::Isometry> current_pose =
513 forward_kinematics(model, particles[i], target_link_name, source_link_name);
514 Scalar fitness_value = homogeneous_error(current_pose, desired_pose).squaredNorm();
515
516 // Update the best position of the particle
517 if (fitness_value < fitness_values[i]) {
518 fitness_values[i] = fitness_value;
519 }
520
521 // Update the best global position
522 if (fitness_value < best_global_fitness) {
523 best_global_fitness = fitness_value;
524 best_global_position = particles[i];
525 }
526 }
527
528 // Update the particles' velocities and positions
529 for (int i = 0; i < num_particles; ++i) {
530 // Update the velocity
531 velocities[i] =
532 omega * velocities[i]
533 + c1 * Eigen::Matrix<Scalar, nq, 1>::Random().cwiseProduct(particles[i] - best_global_position)
534 + c2 * Eigen::Matrix<Scalar, nq, 1>::Random().cwiseProduct(best_global_position - particles[i]);
535
536 // Update the position
537 particles[i] += velocities[i];
538 }
539
540 // Check if the error is within tolerance
541 if (best_global_fitness < options.tolerance) {
542 break;
543 }
544 }
545
546 // Return the best global position
547 return best_global_position;
548 }

561 template <typename Scalar, int nq>
562 Eigen::Matrix<Scalar, nq, 1> inverse_kinematics(Model<Scalar, nq>& model,
563 const std::string& target_link_name,
564 const std::string& source_link_name,
565 const Eigen::Transform<Scalar, 3, Eigen::Isometry>& desired_pose,
566 const Eigen::Matrix<Scalar, nq, 1> q0,
567 const InverseKinematicsOptions<Scalar, nq>& options) {
568 switch (options.method) {
569 case InverseKinematicsMethod::NLOPT:
